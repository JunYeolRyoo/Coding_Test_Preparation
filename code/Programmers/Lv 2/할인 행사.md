``` python
def solution(want, number, discount):
    answer = 0
    l,r = 0,9
    items = {}
    wantnumlist = []
    
    for i in range(len(want)):
        wantnumlist.append((want[i],number[i]))
        
    for i in range(10):
        if discount[i] not in items: items[discount[i]] = 1
        else: items[discount[i]] += 1
        
    while r < len(discount):
        for elem in wantnumlist:
            if elem[0] not in items or items[elem[0]] < elem[1]: break
        else:
            answer += 1
            
        items[discount[l]] -= 1
        if items[discount[l]] == 0: del items[discount[l]]
        l += 1
        
        if r+1 <= len(discount)-1:
            if discount[r+1] in items: items[discount[r+1]] += 1
            else: items[discount[r+1]] = 1
        r+= 1
    return answer
```

***Time Complexity: O(N)<br>***
***Space Complexity: O(1)***
