``` python
from collections import deque
def solution(maps):
    answer = []
    visited = set()
    queue = deque([])
    
    for i in range(len(maps)):
        for j in range(len(maps[0])):
            if maps[i][j] != "X" and (i,j) not in visited:
                queue.append((i,j))
                visited.add((i,j))
                days = 0
                while queue:
                    x,y = queue.popleft()
                    days += int(maps[x][y])
                    for dx,dy in [(1,0),(-1,0),(0,1),(0,-1)]:
                        if dx+x < 0 or dx+x >= len(maps) or dy+y<0 or dy+y >= len(maps[0]) or (dx+x,dy+y) in visited or maps[dx+x][dy+y] == "X": continue
                        queue.append((dx+x,dy+y))
                        visited.add((dx+x,dy+y))
                answer.append(days)
    
    return [-1] if len(answer) == 0 else sorted(answer)
```

***Time Complexity: O(R * C + k log k) (R<=100, C<=100, k is number of island)<br>***
***Space Complexity: O(RC)***
