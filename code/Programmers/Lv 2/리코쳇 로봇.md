``` python
from collections import deque

def solution(board):
    directions = [(1, 0), (-1, 0), (0, 1), (0, -1)]
    queue = deque([])
    visited = set()
    
    for i in range(len(board)):
        for j in range(len(board[0])):
            if board[i][j] == "R":
                queue.append((i,j,0))
                visited.add((i,j,0))

    while queue:
        x, y, moved = queue.popleft()

        if board[x][y] == "G":
            return moved

        for dx, dy in directions:
            nx, ny = x, y

            while True:
                tx, ty = nx + dx, ny + dy

                if tx < 0 or tx >= len(board) or ty < 0 or ty >= len(board[0]) or board[tx][ty] == "D":
                    break

                nx, ny = tx, ty

            if (nx, ny) not in visited:
                visited.add((nx, ny))
                queue.append((nx, ny, moved + 1))
    return -1
```

***Time Complexity: O(R * C * (R + C)) (R <= 100, C <= 100) <br>***
***Space Complexity: O(R * C)***
