``` python
from collections import deque
class Solution:
    def orangesRotting(self,grid):
        minute = -1
        fresh = 0
        current = deque()

        for r in range(len(grid)):
            for c in range(len(grid[0])):
                if grid[r][c] == 1: fresh += 1
                elif grid[r][c] == 2: current.append((r,c))
        
        if fresh == 0: return 0
        
        while current:
            for _ in range(len(current)):
                r,c = current.popleft()
                for nr,nc in ((r+1,c),(r-1,c),(r,c+1),(r,c-1)):
                    if nr < 0 or nr == len(grid) or nc < 0 or nc == len(grid[0]) or grid[nr][nc] != 1: continue
                    grid[nr][nc] = 2
                    current.append((nr,nc))
                    fresh -= 1
            minute += 1
        
        if fresh == 0: return minute
        else: return -1
```

**Time Complexity: O(row*column)** <br>
**Space Complexity: O(row*column)**
