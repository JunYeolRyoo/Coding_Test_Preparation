``` python
class Solution:
    def helper(self,root):
        if root == None: return None
        l = self.helper(root.left)
        r = self.helper(root.right)

        if l == False or r == False: return False
        if l == None and r == None: return [root.val,root.val]
        elif l == None and r != None:
            if root.val >= r[0]: return False
            return [root.val,r[1]]
        elif l != None and r == None:
            if root.val <= l[1]: return False
            return [l[0],root.val]
        else:
            if not(root.val>l[1] and root.val<r[0]): return False
            return [l[0],r[1]]
        
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        result = self.helper(root)
        if result == False: return False
        else: return True
```

**Time Complexity: O(N)<br>**
**Space Complexity: O(h) (Recursion)**
