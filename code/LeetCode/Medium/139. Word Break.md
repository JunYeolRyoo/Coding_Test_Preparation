class Solution:
    def __init__(self):
        self.charDict = {}
        self.nonavail = {}

    def dfs(self,ind,s):
        if ind >= len(s):
            return True
    
        if s[ind] in self.charDict:
            for word in self.charDict[s[ind]]:
                if len(word) > len(s) - ind:
                    continue
                if word == s[ind:ind+len(word)] and (ind+len(word),len(s)-1) not in self.nonavail:
                    res = self.dfs(ind+len(word),s)
                    if res == True:
                        return True
                self.nonavail[(ind,len(s)-1)] = 1
            return False
        else:
            self.nonavail[(ind,len(s)-1)] = 1
            return False

    def wordBreak(self, s, wordDict):
        for word in wordDict:
            if word[0] not in self.charDict:
                self.charDict[word[0]] = [word]
            else:
                self.charDict[word[0]].append(word)
        
        return self.dfs(0,s)
