class TimeMap:
    def __init__(self):
        self.timedict = {}
        self.dict = {}

    def set(self, key: str, value: str, timestamp: int) -> None:
        if key not in self.dict:
            self.dict[key] = [value]
            self.timedict[key] = [timestamp]
        else:
            self.dict[key].append(value)
            self.timedict[key].append(timestamp)

    def get(self, key: str, timestamp: int) -> str:
        if key not in self.dict: return ""
        res = self.binarySearch(self.timedict[key],timestamp)
        if res == None: return ""
        return self.dict[key][res]
    
    def binarySearch(self,l,timestamp):
        res = None
        small= 0
        big = len(l) - 1
        while small <= big:
            mid = (small+big)//2
            if l[mid] == timestamp:
                return mid
            elif l[mid] < timestamp:
                res = mid
                small = mid + 1
            else:
                big = mid - 1
        return res
