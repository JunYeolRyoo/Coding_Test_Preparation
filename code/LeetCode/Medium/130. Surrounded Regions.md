``` python
class Solution:
    def solve(self, board: List[List[str]]) -> None:
        #Do not return anything, modify board in-place instead.
        visited = {}
        def rec(board,m,n):
            if m < 0 or m > len(board)-1 or n < 0 or n > len(board[0])-1: return None
            if (m,n) in visited or board[m][n] == "X": return None
            
            visited[(m,n)] = 1

            rec(board,m+1,n)
            rec(board,m-1,n)
            rec(board,m,n+1)
            rec(board,m,n-1)

        for r in range(len(board)):
            if board[r][0] == "O" and (r,0) not in visited: rec(board,r,0)
            if board[r][len(board[0])-1] == "O" and (r,len(board[0])-1) not in visited: rec(board,r,len(board[0])-1)

        for c in range(len(board[0])):
            if board[0][c] == "O" and (0,c) not in visited: rec(board,0,c)
            if board[len(board)-1][c] == "O" and (len(board)-1,c) not in visited: rec(board,len(board)-1,c)
        
        for r in range(len(board)):
            for c in range(len(board[0])):
                if board[r][c] == "O" and (r,c) not in visited:
                    board[r][c] = "X"
```

**Time complexity: O(mn)**<br>
**Space complexity: O(mn)**
