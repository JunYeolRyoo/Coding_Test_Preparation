``` python
class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        res = 0
        visited = {}
        direction = [[1,0],[0,1],[-1,0],[0,-1]]

        for m in range(len(grid)):
            for n in range(len(grid[0])):
                if grid[m][n] == "1":
                    visited[(m,n)] = 0
        def helper(cell):
            visited[cell] = 1
            for d in direction:
                if cell[0] + d[0] < 0 or cell[0] + d[0] >= len(grid) or cell[1] + d[1] < 0 or cell[1] + d[1] >= len(grid[0]): continue
                if grid[cell[0]+d[0]][cell[1]+d[1]] == "0" or visited[(cell[0]+d[0],cell[1]+d[1])] == 1: continue
                helper((cell[0]+d[0],cell[1]+d[1]))

        for key in visited:
            if visited[key] == 0:
                helper(key)
                res += 1
        return res
```

**Time Complexity: O(m*n)<br>**
**Space Complexity: O(m*n)**
