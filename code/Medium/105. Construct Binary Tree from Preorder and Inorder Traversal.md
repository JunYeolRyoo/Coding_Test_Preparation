``` python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
import collections
class Solution:
    def __init__(self):
        self.queue = None
        self.visited = {}
        self.index = {}
    def helper(self,pre,inor):
        if len(pre) == 0: return None
        root = pre.popleft()
        self.visited[root] = True
        left, right = None,None
        if self.index[root] - 1 > -1 and not self.visited[inor[self.index[root]-1]]:
            left = self.helper(pre,inor)
        if self.index[root] + 1 < len(inor) and not self.visited[inor[self.index[root]+1]]:
            right = self.helper(pre,inor)
        node = TreeNode(root,left,right)
        return node

    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        self.queue = collections.deque(preorder)
        for i in preorder:
            self.visited[i] = False
        for i,j in enumerate(inorder):
            self.index[j] = i
        return self.helper(self.queue,inorder)
```
**Time Complexity: O(N)<br>**
**Space Complexity: O(N)**
