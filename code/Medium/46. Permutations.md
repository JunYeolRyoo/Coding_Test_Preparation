``` python
class Solution:
    def permute(self, nums):
        if len(nums) == 1: return [[nums[0]]]

        def dfs(li):
            if len(li) == 1: return [[li[0]]]   # [[x]]
            recRet = []
            for i in range(len(li)):
                copied = li.copy()
                poped = copied.pop(i)
                res = dfs(copied)
                for z in res:
                    z.append(poped)
                recRet.extend(res)
            return recRet
        return dfs(nums)
```
