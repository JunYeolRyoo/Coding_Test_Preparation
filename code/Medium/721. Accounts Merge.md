class UnionFind:
    def __init__(self,n):
        self.parent = [i for i in range(n)]
        self.rank = [1] * n
    
    def find(self,elem):
        while self.parent[elem] != elem:
            elem = self.parent[elem]
        return elem
    
    def union(self,elem1,elem2):
        par1 = self.find(elem1)
        par2 = self.find(elem2)
        if par1 != par2:
            if self.rank[par1] < self.rank[par2]:
                self.parent[par1] = par2
                self.rank[par2] += self.rank[par1]
            else:
                self.parent[par2] = par1
                self.rank[par1] += self.rank[par2]
            return True
        else:
            return False

class Solution:
     def accountsMerge(self, accounts):
        uf = UnionFind(len(accounts))

        emailDict = {}
        for i in range(len(accounts)):
            for email in accounts[i][1:]:
                 if email in emailDict:
                      uf.union(emailDict[email],i)
                 else:
                      emailDict[email] = i
        finaldict = {}
        for key,val in emailDict.items():
            findRes = uf.find(val)
            if findRes not in finaldict:
                finaldict[findRes] = [key]
            else:
                finaldict[findRes].append(key)
        res = []
        for i in finaldict:
            res.append([accounts[i][0]] + sorted(finaldict[i]))
        return res
