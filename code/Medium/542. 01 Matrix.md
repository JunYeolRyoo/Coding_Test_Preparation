``` python
class Solution:
    def updateMatrix(self, mat: List[List[int]]) -> List[List[int]]:
        res = [[None for i in range (len(mat[0]))] for j in range (len(mat))]
        stack = []
        nextStack = []
        direction = [[-1,0],[0,1],[1,0],[0,-1]]
        dis = 1

        for m in range(len(mat)):
            for n in range(len(mat[0])):
                if mat[m][n] == 0:
                    res[m][n] = 0
                    stack.append([m,n])
        
        while True:
            if len(stack) == 0: break
            newStack = []
            for item in stack:
                m,n = item[0], item[1]
                for row,col in direction:
                    if m + row < 0 or m + row >= len(mat) or n + col < 0 or n + col >= len(mat[0]):
                        continue
                    if res[m+row][n+col] == None:
                        res[m+row][n+col] = dis
                        newStack.append([m+row,n+col])
            dis += 1
            stack = newStack
        return res
```

**Time Complexity: O(m*n)<br>**
**Space Complexity: O(m*n)**
