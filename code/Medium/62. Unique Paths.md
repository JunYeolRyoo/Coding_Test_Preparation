``` python
class Solution:
    def dfs(self,row,col,m,n,dic):
        if row >= m-1 and col >= n-1:
            return 1

        right,down = 0,0
        if (row,col) in dic:
            return dic[(row,col)]
        if row <= m-2:
            down = self.dfs(row+1,col,m,n,dic)
        if col <= n-2:
            right = self.dfs(row,col+1,m,n,dic)
        dic[(row,col)] = down + right
        return down + right
        
    def uniquePaths(self, m: int, n: int) -> int:
        dic = {}
        return self.dfs(0,0,m,n,dic)
```

**Time Complexity: O(m * n)<br>**
**Space Complexity: O(m * n)**
