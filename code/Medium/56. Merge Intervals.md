class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        intervals.sort(key=lambda x: x[0])
        res = []

        prev = intervals[0]
        for ind in range(1,len(intervals)):
            if intervals[ind][0] > prev[1]:
                res.append(prev)
                prev = intervals[ind]
            else:
                if intervals[ind][1] > prev[1]:
                    prev = [prev[0],intervals[ind][1]]
        
        if len(res) == 0 or res[len(res)-1] != prev: res.append(prev)

        return res
