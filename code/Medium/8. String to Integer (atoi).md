``` python
class Solution:
    def myAtoi(self, s):
        result = "0"
        checkWhiteSpace = True
        negative = False

        for c in s:
            if checkWhiteSpace:
                if c == " ": continue
                elif c == "-" or c == "+":
                    if c == "-": negative = True
                    checkWhiteSpace = False
                    continue
                checkWhiteSpace = False
            if not c.isdigit(): break
            result += c
        
        result = int(result)
        if negative: result *= -1
        if result <= pow(-2,31): result = pow(-2,31)
        elif result >= pow(2,31)-1: result = pow(2,31)-1
        return result
```
